    /**
     * Calcule le nombre de jours qui se chevauchent entre deux plages de dates
     */
    private calculateOverlappingDays(
        start1: Date, end1: Date,
        start2: Date, end2: Date
    ): number {
        const overlapStart = start1 < start2 ? start2 : start1;

function mapShiftType(shiftType: string): ShiftType {
    const mapping: Record<string, ShiftType> = {
        'GUARD_24H': 'GARDE_24H',

        'CONSULTATION': 'CONSULTATION',
        'REST': 'REPOS',
        'LEAVE': 'CONGE'
    };
    return mapping[shiftType] || 'VACATION';
}
